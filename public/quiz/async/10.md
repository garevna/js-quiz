  const results = []

  const promise = num => new Promise(resolve => setTimeout(() => resolve(num), 500 * num))

  promise(5).then(res => results.push(res))

  promise(4).then(res => results.push(res))
  results.push(await promise(2))
  results.push(await promise(3))

  return results
}
