function User ( name ) {
    var userName = name || "User"
    var postList = []
    Object.defineProperties ( this, {
        name: {
            get () {
                return userName
            },
            set ( newName ) {
                userName = newName
            }
        },
        posts: {
            get () {

                    JSON.stringify ( postList )

            },

                    postList.push ({
                        title: postData.header || "empty",
                        text: postData.text || "empty",
                    }) : null
            }
        }
    })
}
