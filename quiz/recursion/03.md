function addElem ( tags ) {
    var tag = tags.shift()
    var elem = document.body.appendChild (
        document.createElement ( tag.name )
    )
    for ( var x in tag.attrs ) {
        elem [ x ] = tag.attrs [ x ]
    }
    while ( tags.length ) addElem ()
}
